# 项目初始化脚本

## 快速开始

### 1. 克隆项目并初始化

```bash
# 创建项目目录
mkdir storyflow && cd storyflow

# 初始化git
git init

# 创建项目结构
mkdir -p frontend backend ai-service kubernetes
```

### 2. 后端初始化

```bash
cd backend

# 初始化Node项目
npm init -y

# 安装依赖
npm install express cors dotenv pg redis axios typescript ts-node @types/node @types/express

# 创建目录结构
mkdir -p src/{routes,controllers,services,models,middleware,utils,db}
mkdir -p migrations

# 创建tsconfig.json
cat > tsconfig.json << 'EOF'
{
  "compilerOptions": {
    "target": "ES2020",
    "module": "commonjs",
    "lib": ["ES2020"],
    "outDir": "./dist",
    "rootDir": "./src",
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true,
    "resolveJsonModule": true,
    "declaration": true,
    "declarationMap": true,
    "sourceMap": true
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules", "dist"]
}
EOF

# 创建.env.example
cat > .env.example << 'EOF'
NODE_ENV=development
PORT=3000
DATABASE_URL=postgresql://storyflow:password@localhost:5432/storyflow
REDIS_URL=redis://localhost:6379
JWT_SECRET=your_jwt_secret_key_here
JWT_EXPIRES_IN=7d
CLAUDE_API_KEY=sk-ant-...
AWS_ACCESS_KEY_ID=your_access_key
AWS_SECRET_ACCESS_KEY=your_secret_key
AWS_S3_BUCKET=storyflow-content
AWS_REGION=us-east-1
EOF

# 创建Dockerfile
cat > Dockerfile << 'EOF'
FROM node:18-alpine

WORKDIR /app

COPY package*.json ./
RUN npm ci --only=production

COPY src ./src
COPY tsconfig.json ./

RUN npm run build

EXPOSE 3000

CMD ["node", "dist/index.js"]
EOF

# 创建package.json脚本
npm set-script dev "ts-node src/index.ts"
npm set-script build "tsc"
npm set-script start "node dist/index.js"
npm set-script test "jest"
```

### 3. 前端初始化

```bash
cd ../frontend

# 使用Vite创建Vue 3项目
npm create vite@latest . -- --template vue-ts

# 安装额外依赖
npm install pinia axios tailwindcss postcss autoprefixer

# 初始化Tailwind
npx tailwindcss init -p

# 创建.env.example
cat > .env.example << 'EOF'
VITE_API_URL=http://localhost:3000/api
VITE_APP_NAME=StoryFlow
EOF
```

### 4. AI服务初始化

```bash
cd ../ai-service

# 创建Python虚拟环境
python3 -m venv venv
source venv/bin/activate  # macOS/Linux
# 或 venv\Scripts\activate  # Windows

# 创建requirements.txt
cat > requirements.txt << 'EOF'
fastapi==0.104.1
uvicorn==0.24.0
pydantic==2.5.0
anthropic==0.7.0
psycopg2-binary==2.9.9
python-dotenv==1.0.0
redis==5.0.1
EOF

# 安装依赖
pip install -r requirements.txt

# 创建.env.example
cat > .env.example << 'EOF'
CLAUDE_API_KEY=sk-ant-...
DATABASE_URL=postgresql://storyflow:password@localhost:5432/storyflow
REDIS_URL=redis://localhost:6379
EOF

# 创建Dockerfile
cat > Dockerfile << 'EOF'
FROM python:3.11-slim

WORKDIR /app

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY src ./src

EXPOSE 8000

CMD ["uvicorn", "src.main:app", "--host", "0.0.0.0", "--port", "8000"]
EOF
```

### 5. Docker Compose启动

```bash
cd ..

# 创建docker-compose.yml (见设计文档)

# 启动所有服务
docker-compose up -d

# 查看日志
docker-compose logs -f

# 停止服务
docker-compose down
```

### 6. 数据库初始化

```bash
# 连接到PostgreSQL
psql postgresql://storyflow:password@localhost:5432/storyflow

# 执行初始化SQL (见下面的SQL脚本)
```

## 数据库初始化SQL

```sql
-- 创建users表
CREATE TABLE users (
  id BIGSERIAL PRIMARY KEY,
  email VARCHAR(255) UNIQUE NOT NULL,
  phone VARCHAR(20),
  password_hash VARCHAR(255) NOT NULL,
  username VARCHAR(100) UNIQUE NOT NULL,
  avatar_url TEXT,
  language VARCHAR(10) DEFAULT 'zh-TW',
  subscription_tier VARCHAR(50) DEFAULT 'free',
  subscription_expires_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  deleted_at TIMESTAMP
);

CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_username ON users(username);

-- 创建stories表
CREATE TABLE stories (
  id BIGSERIAL PRIMARY KEY,
  title VARCHAR(255) NOT NULL,
  slug VARCHAR(255) UNIQUE NOT NULL,
  category VARCHAR(50) NOT NULL,
  language VARCHAR(10) NOT NULL,
  content TEXT NOT NULL,
  summary TEXT NOT NULL,
  cover_url TEXT,
  author_name VARCHAR(100) DEFAULT 'AI-StoryFlow',
  word_count INT NOT NULL,
  chapter_count INT DEFAULT 1,
  views INT DEFAULT 0,
  likes INT DEFAULT 0,
  comments_count INT DEFAULT 0,
  rating DECIMAL(3,2) DEFAULT 0,
  ai_model VARCHAR(100),
  prompt_version INT,
  generated_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  published_at TIMESTAMP
);

CREATE INDEX idx_stories_category ON stories(category);
CREATE INDEX idx_stories_language ON stories(language);
CREATE INDEX idx_stories_published_at ON stories(published_at);
CREATE INDEX idx_stories_views ON stories(views);

-- 创建story_chapters表
CREATE TABLE story_chapters (
  id BIGSERIAL PRIMARY KEY,
  story_id BIGINT NOT NULL REFERENCES stories(id) ON DELETE CASCADE,
  chapter_number INT NOT NULL,
  title VARCHAR(255),
  content TEXT NOT NULL,
  word_count INT,
  unlock_type VARCHAR(50) DEFAULT 'free',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(story_id, chapter_number)
);

CREATE INDEX idx_story_chapters_story_id ON story_chapters(story_id);

-- 创建reading_history表
CREATE TABLE reading_history (
  id BIGSERIAL PRIMARY KEY,
  user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  story_id BIGINT NOT NULL REFERENCES stories(id) ON DELETE CASCADE,
  chapter_number INT,
  scroll_position INT,
  read_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  completed_at TIMESTAMP,
  UNIQUE(user_id, story_id)
);

CREATE INDEX idx_reading_history_user_id ON reading_history(user_id);
CREATE INDEX idx_reading_history_read_at ON reading_history(read_at);

-- 创建user_unlocks表
CREATE TABLE user_unlocks (
  id BIGSERIAL PRIMARY KEY,
  user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  story_id BIGINT NOT NULL REFERENCES stories(id) ON DELETE CASCADE,
  chapter_number INT,
  unlock_method VARCHAR(50),
  unlocked_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(user_id, story_id, chapter_number)
);

CREATE INDEX idx_user_unlocks_user_id ON user_unlocks(user_id);

-- 创建user_likes表
CREATE TABLE user_likes (
  id BIGSERIAL PRIMARY KEY,
  user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  story_id BIGINT NOT NULL REFERENCES stories(id) ON DELETE CASCADE,
  liked_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(user_id, story_id)
);

CREATE INDEX idx_user_likes_user_id ON user_likes(user_id);

-- 创建subscriptions表
CREATE TABLE subscriptions (
  id BIGSERIAL PRIMARY KEY,
  user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  tier VARCHAR(50) NOT NULL,
  price DECIMAL(10,2),
  currency VARCHAR(10) DEFAULT 'USD',
  billing_cycle VARCHAR(50),
  started_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  expires_at TIMESTAMP NOT NULL,
  auto_renew BOOLEAN DEFAULT TRUE,
  payment_method VARCHAR(50)
);

CREATE INDEX idx_subscriptions_user_id ON subscriptions(user_id);
CREATE INDEX idx_subscriptions_expires_at ON subscriptions(expires_at);

-- 创建analytics表
CREATE TABLE analytics (
  id BIGSERIAL PRIMARY KEY,
  date DATE NOT NULL,
  metric_type VARCHAR(100),
  metric_value DECIMAL(15,2),
  story_id BIGINT REFERENCES stories(id),
  UNIQUE(date, metric_type, story_id)
);

CREATE INDEX idx_analytics_date ON analytics(date);
```

## 部署到生产环境

### 1. 构建Docker镜像

```bash
# 构建后端镜像
docker build -t storyflow-backend:latest ./backend

# 构建前端镜像
docker build -t storyflow-frontend:latest ./frontend

# 构建AI服务镜像
docker build -t storyflow-ai:latest ./ai-service
```

### 2. 推送到镜像仓库

```bash
# 登录Docker Hub
docker login

# 标记镜像
docker tag storyflow-backend:latest your-username/storyflow-backend:latest
docker tag storyflow-frontend:latest your-username/storyflow-frontend:latest
docker tag storyflow-ai:latest your-username/storyflow-ai:latest

# 推送镜像
docker push your-username/storyflow-backend:latest
docker push your-username/storyflow-frontend:latest
docker push your-username/storyflow-ai:latest
```

### 3. 部署到Kubernetes

```bash
# 创建命名空间
kubectl create namespace storyflow

# 应用配置
kubectl apply -f kubernetes/configmap.yaml -n storyflow
kubectl apply -f kubernetes/secret.yaml -n storyflow

# 部署服务
kubectl apply -f kubernetes/postgres.yaml -n storyflow
kubectl apply -f kubernetes/redis.yaml -n storyflow
kubectl apply -f kubernetes/backend.yaml -n storyflow
kubectl apply -f kubernetes/frontend.yaml -n storyflow
kubectl apply -f kubernetes/ai-service.yaml -n storyflow

# 查看部署状态
kubectl get pods -n storyflow
kubectl get svc -n storyflow
```

## 常见问题

### Q: 如何修改数据库连接字符串？
A: 修改`.env`文件中的`DATABASE_URL`

### Q: 如何添加新的API端点？
A: 在`backend/src/routes/`中创建新的路由文件，然后在`index.ts`中注册

### Q: 如何生成新的小说？
A: 调用AI服务的`/generate`端点，或使用后台任务定期生成

### Q: 如何监控系统性能？
A: 使用Prometheus + Grafana，配置见`kubernetes/`目录
